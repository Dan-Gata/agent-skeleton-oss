# Utilise la version LTS de Node
FROM node:lts-alpine

# Répertoire de travail dans le conteneur
WORKDIR /app

# Copie seulement les fichiers nécessaires pour installer les dépendances
COPY package.json package-lock.json* ./

# Installe les dépendances (axios, express)
RUN npm install

# Copie tout le code de l’orchestrateur
COPY . .

# Expose le port sur lequel tourne l’application
EXPOSE 3000

# Commande de démarrage de l’application
CMD ["node", "src/index.js"]
